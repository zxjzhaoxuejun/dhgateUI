# DHgate-UI 组件库

# 快速开始

# 1.安装组件库

npm i dhgate-ui

# 2.引用组件库

//全部引入
import 'dhgate-ui/dist/css/index.css'
import MUI from 'dhgate-ui'
Vue.use(MUI)

//按需引用
import 'dhgate-ui/dist/css/demo.css'
import {Demo} from 'dhgate-ui'
Vue.use(Demo)
